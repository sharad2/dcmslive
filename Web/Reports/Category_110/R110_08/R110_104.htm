This Report displays SKU wise informtion of open orders.

Overview

For the specified values of Order Type, Dates, Warehouse Locations ,Vwh ID,Customer,Bucket,PO,Iteration and Reporting Status,
  1. The detail of SKU, Warehouse Location wise.

Algorithm

All SKUs in pickslips are visible to this report except SKUs in:

  • Cancelled Pickslips.
  • Transferred Pickslips.

For finding the UPC CODE, style, color, dimension, sku size, price, number of ordered pieces and dollars, pieces and units in boxes for passed PO, order status (‘IN ORDERED BUCKET’), customer, virtual warehouse and warehouse Location using the table DEM_PICKSLIP and DEM_PICKSLIP_DETAIL table where the pickslip status is 1.

For finding the UPC CODE, style, color, dimension, sku size, price, number of ordered pieces and dollars, pieces and units in boxes for passed PO, order status (except ‘IN ORDERED BUCKET’) , customer/bucket, virtual warehouse and warehouse location using the table PS, PO, BOX, BOXDET and MASTER_SKU where the pickslip reporting status is not null of those status of order.

  9. Iteration(Type-String): This parameter can be used to specify value of Iteration

Query documentation:

The report query is as follows:

 
   WITH Q1 AS
 (SELECT PS.PICKSLIP_ID,
         PD.UPC_CODE,
         PD.PIECES_ORDERED,
         PD.EXTENDED_PRICE,
         PO.DC_CANCEL_DATE,
         PO.START_DATE,
         PS.PICKSLIP_IMPORT_DATE,
         PO.PO_ID AS PO_ID,
         NVL(PS.WAREHOUSE_LOCATION_ID, 'Unknown') AS WAREHOUSE_LOCATION_ID,
         PS.VWH_ID AS VWH_ID
    FROM PS
INNER JOIN PO
      ON PS.CUSTOMER_ID = PO.CUSTOMER_ID
     AND PS.PO_ID = PO.PO_ID
     AND PS.ITERATION = PO.ITERATION
   INNER JOIN PSDET PD
      ON PS.PICKSLIP_ID = PD.PICKSLIP_ID
   WHERE PS.TRANSFER_DATE IS NULL
     AND PD.TRANSFER_DATE IS NULL
     AND ps.vwh_id = :vwh_id    
     AND :WAREHOUSE_LOCATION_ID      
     AND PS.REPORTING_STATUS=:reporting_status
     AND 1 = 2    
     AND po.start_date >= cast(:delivery_start_date as date)  
     AND po.start_date < cast(:delivery_end_date as date) + 1
     AND po.dc_cancel_date >=cast(:dc_cancel_start_date as date)
     AND po.dc_cancel_date < cast(:dc_cancel_end_date as date) + 1
     AND ps.customer_id NOT IN (select c.customer_id from cust c WHERE account_type = 'RTL')
     AND ps.customer_id NOT IN (select c.customer_id from cust c WHERE account_type = 'IDC')
     AND ps.Export_Flag IS NULL
     AND ps.Export_Flag IS NOT NULL
     AND ps.customer_id=:customer_id
     AND ps.po_id=:po_id
     AND ps.bucket_id=:BUCKET_ID
     AND ps.iteration=:iteration
     AND PS.CUSTOMER_DC_ID = :customer_dc_id
     AND ps.CUSTOMER_STORE_ID = :customer_store_id
  UNION ALL
  SELECT DP.PICKSLIP_ID,
         DPD.UPC_CODE,
         DPD.QUANTITY_ORDERED     AS PIECES_ORDERED,
         DPD.EXTENDED_PRICE,
         DP.DC_CANCEL_DATE,
         DP.DELIVERY_DATE,
         DP.PICKSLIP_IMPORT_DATE,
         DP.CUSTOMER_ORDER_ID     AS PO_ID,
         DP.WAREHOUSE_LOCATION_ID,
         DP.VWH_ID
    FROM DEM_PICKSLIP DP
   INNER JOIN DEM_PICKSLIP_DETAIL DPD
      ON DP.PICKSLIP_ID = DPD.PICKSLIP_ID
   WHERE 1=1
      AND DP.vwh_id = :vwh_id
   
      AND :WAREHOUSE_LOCATION_ID      
      AND DP.PS_STATUS_ID = 1
      AND 1=2
      AND DP.delivery_date >= cast(:delivery_start_date as date)
      AND DP.delivery_date < cast(:delivery_end_date as date) + 1
      AND DP.dc_cancel_date >=cast(:dc_cancel_start_date as date)
      AND DP.dc_cancel_date < cast(:dc_cancel_end_date as date) + 1
      AND DP.customer_id NOT IN (select c.customer_id from cust c WHERE account_type = 'RTL')
      AND DP.customer_id NOT IN (select c.customer_id from cust c WHERE account_type = 'IDC')
      AND DP.Export_Flag IS NULL
      AND DP.Export_Flag IS NOT NULL
      AND DP.customer_id=:customer_id
      AND DP.customer_order_id=:po_id
      AND DP.customer_dist_center_id = :customer_dc_id
      AND DP.CUSTOMER_STORE_ID = :customer_store_id
      AND 1=2 ),
Q2 AS
 (SELECT ROW_NUMBER() OVER(PARTITION BY Q1.PICKSLIP_ID, Q1.UPC_CODE ORDER BY Q1.PICKSLIP_ID, Q1.UPC_CODE) AS PICKSLIP_SEQUENCE,
         Q1.PICKSLIP_ID,
         Q1.UPC_CODE,
         MS.STYLE,
         MS.COLOR,
         MS.DIMENSION,
         MS.SKU_SIZE,
         Q1.PIECES_ORDERED,
         Q1.EXTENDED_PRICE,
         Q1.DC_CANCEL_DATE,
         Q1.START_DATE,
         Q1.PICKSLIP_IMPORT_DATE,
         Q1.PO_ID,
         Q1.WAREHOUSE_LOCATION_ID,
         Q1.VWH_ID
    FROM Q1
    LEFT OUTER JOIN PS_VAS PV
      ON Q1.PICKSLIP_ID = PV.PICKSLIP_ID
    LEFT OUTER JOIN MASTER_SKU MS
      ON Q1.UPC_CODE = MS.UPC_CODE),
ALL_PSDATA AS
 (SELECT Q1.PICKSLIP_ID,
         Q1.UPC_CODE,
         MAX(Q1.STYLE) AS STYLE,
         MAX(Q1.COLOR) AS COLOR,
         MAX(Q1.DIMENSION) AS DIMENSION,
         MAX(Q1.SKU_SIZE) AS SKU_SIZE,
         SUM(Q1.PIECES_ORDERED) AS PIECES_ORDERED,
         SUM(Q1.PIECES_ORDERED * Q1.EXTENDED_PRICE) AS DOLLARS_ORDERED,
         SUM(Q1.EXTENDED_PRICE) AS EXTENDED_PRICE,
         MAX(Q1.DC_CANCEL_DATE) AS DC_CANCEL_DATE,
         MAX(Q1.START_DATE) AS DELIVERY_DATE,
         MAX(Q1.PICKSLIP_IMPORT_DATE) AS IMPORT_DATE,
         MAX(Q1.PO_ID) AS PO_ID,
         MAX(Q1.WAREHOUSE_LOCATION_ID) AS WAREHOUSE_LOCATION_ID,
         MAX(Q1.VWH_ID) AS VWH_ID
    FROM Q2 Q1
   WHERE PICKSLIP_SEQUENCE = 1
   GROUP BY Q1.PICKSLIP_ID, Q1.UPC_CODE),
BOX_PICKSLIPS AS
 (SELECT BD.PICKSLIP_ID AS PICKSLIP_ID,
         BD.UPC_CODE,
         SUM(BD.CURRENT_PIECES / NVL(SKU.PIECES_PER_PACKAGE, 1)) AS UNITS_SHIPPED,
         SUM(BD.CURRENT_PIECES * BD.EXTENDED_PRICE) AS SHIPPED_DOLLARS,
         SUM(BD.CURRENT_PIECES) AS CURRENT_PIECES
    FROM BOX B
   INNER JOIN BOXDET BD
      ON B.UCC128_ID = BD.UCC128_ID
     AND B.PICKSLIP_ID = BD.PICKSLIP_ID
   INNER JOIN PS
      ON B.PICKSLIP_ID = PS.PICKSLIP_ID
   INNER JOIN PO
      ON PS.CUSTOMER_ID = PO.CUSTOMER_ID
     AND PS.PO_ID = PO.PO_ID
     AND PS.ITERATION = PO.ITERATION
    LEFT OUTER JOIN MASTER_SKU SKU
      ON BD.UPC_CODE = SKU.UPC_CODE
   WHERE BD.STOP_PROCESS_DATE IS NULL
     AND B.STOP_PROCESS_DATE IS NULL
     AND PS.TRANSFER_DATE IS NULL
     AND ps.vwh_id = :vwh_id    
     AND :WAREHOUSE_LOCATION_ID      
     AND PS.REPORTING_STATUS=:reporting_status
     AND 1 = 2    
     AND po.start_date >= cast(:delivery_start_date as date)  
     AND po.start_date < cast(:delivery_end_date as date) + 1
     AND po.dc_cancel_date >=cast(:dc_cancel_start_date as date)
     AND po.dc_cancel_date < cast(:dc_cancel_end_date as date) + 1
     AND ps.customer_id NOT IN (select c.customer_id from cust c WHERE account_type = 'RTL')
     AND ps.customer_id NOT IN (select c.customer_id from cust c WHERE account_type = 'IDC')
     AND ps.Export_Flag IS NULL
     AND ps.Export_Flag IS NOT NULL
     AND ps.customer_id=:customer_id
     AND ps.po_id=:po_id
     AND ps.bucket_id=:BUCKET_ID
     AND ps.iteration=:iteration
     AND PS.CUSTOMER_DC_ID = :customer_dc_id
     AND ps.CUSTOMER_STORE_ID = :customer_store_id
     AND BD.CURRENT_PIECES > 0
   GROUP BY BD.UPC_CODE, BD.PICKSLIP_ID)
SELECT PSDATA.UPC_CODE AS UPC_CODE,
       MAX(PSDATA.STYLE) AS STYLE,
       MAX(PSDATA.COLOR) AS COLOR,
       MAX(PSDATA.DIMENSION) AS DIMENSION,
       MAX(PSDATA.SKU_SIZE) AS SKU_SIZE,
       MAX(PSDATA.EXTENDED_PRICE) AS PRICE,
       SUM(PSDATA.PIECES_ORDERED) AS PIECES_ORDERED,
       SUM(PSDATA.DOLLARS_ORDERED) AS DOLLARS_ORDERED,
       SUM(BP.CURRENT_PIECES) AS PIECES_IN_BOXES,
       ROUND(SUM(BP.UNITS_SHIPPED)) AS UNITS_SHIPPED,
       SUM(BP.SHIPPED_DOLLARS) AS SHIPPED_DOLLARS,
       PSDATA.VWH_ID AS VWH_ID,
       PSDATA.WAREHOUSE_LOCATION_ID AS WAREHOUSE_LOCATION_ID
  FROM ALL_PSDATA PSDATA
  LEFT OUTER JOIN BOX_PICKSLIPS BP
    ON BP.UPC_CODE = PSDATA.UPC_CODE
   AND BP.PICKSLIP_ID = PSDATA.PICKSLIP_ID
 GROUP BY PSDATA.UPC_CODE, PSDATA.VWH_ID, PSDATA.WAREHOUSE_LOCATION_ID

select
 pd.pieces_ordered * pd.extended_price as Dollars_ordered,
 pd.pieces_ordered as Quantity_ordered,
 pd.upc_code
  from ps ps
 inner join psdet pd
    on ps.pickslip_id = pd.pickslip_id
 inner join po po
    on po.po_id = ps.po_id
   and po.customer_id = ps.customer_id
   and po.iteration = ps.iteration
 inner join master_sku msku
    on pd.upc_code = msku.upc_code
 where msku.style =:style
   and msku.color =:color 
   and msku.dimension =:dimension 
   and msku.sku_size =:sku_size 
   and ps.warehouse_location_id =:warehouse_location_id 
   and ps.vwh_id =:vwh_id 
   and po.start_date >=:start_date 
   and po.start_date <=:start_date
   and po.dc_cancel_date >=:dc_cancel_date 
   and po.dc_cancel_date <=:dc_cancel_date 
   and ps.reporting_status =:reporting_status

TO VALIDATE BOX PIECES PLEASE RUN FOLLOWING QUERY
  
SELECT PS.VWH_ID AS VWH_ID,
       PS.WAREHOUSE_LOCATION_ID AS WAREHOUSE_LOCATION_ID,
       BD.UPC_CODE,
       MAX(PS.REPORTING_STATUS) AS REPORTING_STATUS,
       SUM(BD.CURRENT_PIECES / NVL(SKU.PIECES_PER_PACKAGE, 1)) AS UNITS_SHIPPED,
       SUM(BD.CURRENT_PIECES) AS CURRENT_PIECES
  FROM BOXDET BD
 INNER JOIN PS PS
    ON BD.PICKSLIP_ID = PS.PICKSLIP_ID
  LEFT OUTER JOIN MASTER_SKU SKU
    ON BD.UPC_CODE = SKU.UPC_CODE
 WHERE BD.STOP_PROCESS_DATE IS NULL
   AND BD.CURRENT_PIECES > 0
   AND PS.WAREHOUSE_LOCATION_ID =:warehouse_location_id
   AND PS.VWH_ID =:vwh_id
   AND PS.REPORTING_STATUS = :rerporting_status
   AND PS.PO_ID = :PO_Id
   AND BD.UPC_CODE =:SKU_id
 GROUP BY BD.UPC_CODE, PS.VWH_ID, PS.WAREHOUSE_LOCATION_ID
